FROM DrMedicationRecomInfo dmri
JOIN MedicationRecom dmr ON dmri.recommendId=dmr.id
JOIN RecomAuditLog ral ON dmri.recommendId=ral.recomId
WHERE dmr.doctorId = ?1
AND dmr.status=0
AND dmr.expireAt < now()
AND dmr.type!=3
AND dmr.forceRecom !=0
AND ral.status=1
